function compareQueues(queue1, queue2){ if(queue1 && queue2){ var curr1 = queue1.head; var curr2 = queue2.head; while(curr1 && curr2){ if(curr1.val !== curr2.val){ return false; } curr1 = curr1.next; curr2 = curr2.next; } return curr1 === curr2; } return false; } // console.log( compareQueues(queue1, queue2) );
